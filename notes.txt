create table emp_at
 (
   empno number(4,0),
   ename varchar2(10) not null,
   job varchar2(9) default 'Clerk' check (job in ('Manager', 'Clerk', 'OfficeAssistant')),
   mgr number(4,0),
   hiredate date,
   sal number(7,2),
   comm number(7,2),
   deptno number(2),
   constraint pk_emp primary key(empno),
   constraint fk_deptno foreign key(deptno) references dept_at(deptno) 
 );

desc emp_at;
